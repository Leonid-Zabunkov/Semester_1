### **Вставками**

```cpp
void insertion_sort(int arr[], int n) {
    for (int j = 0; j < n; j++) {
        int k = arr[j];
        int i = j - 1;
        while (i >= 0 && arr[i] > k) {
            arr[i + 1] = arr[i];
            i--;
        }
        arr[i + 1] = k;
    }
}
```

### **Quick Sort**

#### Разбиение Хоара

```cpp
void quick_sort(int arr[], int l, int r) {
    int i = l;
    int j = r;
    int pivot = arr[(l + r) / 2];
    while (i <= j) {
        while (arr[i] < pivot) {
            i++;
        }
        while (arr[j] > pivot) {
            j--;
        }
        if (i <= j) {
            swap(arr[i], arr[j]);
            i++;
            j--;
        }
    }
    if (i < r) {
        quick_sort(arr, i, r);
    }
    if (l < j) {
        quick_sort(arr, l, j);
    }
}
```

#### Разбиение Ломуто

```cpp
int lomuto_partition(int arr[], int l, int r) {
    int pivot = arr[r];
    int i = l;
    for (int j = l; j < r; j++) {
        if (arr[j] <= pivot) {
            swap(arr[j], arr[i]);
            i++;
        }
    }
    swap(arr[i], arr[r]);
    return i;
}

void lomuto_quick_sort(int arr[], int l, int r) {
    if (l < r) {
        int q = lomuto_partition(arr, l, r);
        lomuto_quick_sort(arr, l, q - 1);
        lomuto_quick_sort(arr, q + 1, r);
    }
}
```

```cpp
int quick_select(int arr[], int l, int r, int k) {
    if (l == r) return arr[l];
    int i = l;
    int j = r;
    int pivot = arr[(l + r) / 2];
    while (i <= j) {
        while (arr[i] < pivot) {
            i++;
        }
        while (arr[j] > pivot) {
            j--;
        }
        if (i <= j) {
            swap(arr[i], arr[j]);
            i++;
            j--;
        }
    }
    if (k <= j) {
        quick_select(arr, l, j, k);
    }
    if (k >= i) {
        quick_select(arr, i, r, k);
    }
    return arr[k];
}
```

### **Merge Sort**

```cpp
void merge(int arr[], int l, int mid, int r) {
    int res[r - l];
    int i1 = 0, i2 = 0;
    while (i1 + l < mid && i2 + mid < r) {
        if (arr[i1 + l] < arr[i2 + r]) {
            res[i1 + i2] = arr[i1 + l];
            i1++;
        } else {
            res[i1 + i2] = arr[i2 + mid];
            i2++;
        }
    }
    while (i1 + l < mid) {
        res[i1 + i2] = arr[i1 + l];
        i1++;
    }
    while (i2 + mid < r) {
        res[i1 + i2] = arr[i2 + mid];
        i2++;
    }
    for (int i = 0; i < i1 + i2; i++) {
        arr[l + i] = res[i];
    }
}

void merge_sort(int arr[], int l, int r) {
    if (l <= r) {
        int mid = (l + r) / 2;
        merge_sort(arr, l, mid);
        merge_sort(arr, mid, r);
        merge(arr, l, mid, r);
    }
}
```

```cpp
void stable_counting_sort(int arr[], int n, int k) {
    int c[k];// если не знаем диапазон чисел, то линейно ищем max & min в массвиве перед сортировкой
    int tmp[n];//т.к. min может быть отрицательным, то из arr[i] вычитаем min, когда добавляем в c
        //и прибавляем min, когда записываем в tmp
    for (int i = 0; i < k; i++) {
        c[i] = 0;
    }
    for (int i = 0; i < n; i++) { //!!!Учесть, что размер массива должен быть соответствующим
        c[arr[i]]++; //вычесть min, если min < 0
    }
    for (int i = 1; i < k; i++) {
        c[i] += c[i - 1];
    }
    for (int i = n - 1; i >= 0; i--) {
        int x = arr[i];
        tmp[c[x]] = x;
        c[x]--;
    }
    for (int i = 0; i < n; i++) {
        arr[i] = tmp[i]; //прибавить min, если min < 0
    }
}
```

### **Цифровая**

#### LSD

**Идет с последнего(меньшего разряда)**

```cpp
void lsd_radix_sort(int arr[], int n, int k) {
    const int base = 1 << 4; //сортируем по k разрядам в 16-ной системе счисления(2**b, где b = 4)
    for (int j = 0; j < k; j++) { // можно написать в 10-ной системе, но тогда будет медленно
        int c[base] = {0}; 
        int tmp[n];
        for (int i = 0; i < n; i++) {
            int digit = (arr[i] >> j)  & (base - 1);    //arr[i] / 2 ** b and arr[i] % 2**b
            c[digit]++;
        }
        for (int i = 1; i < base; i++) {
            c[i] += c[i - 1];
        }
        for (int i = n - 1; i >= 0; i--) {
            int digit = (arr[i] >> j)  & (base - 1);  
            tmp[c[digit] - 1] = arr[i];
            c[digit]--;
        }
        for (int i = 0; i < n; i++) {
            arr[i] = tmp[i];
        }
    }
}
```

#### MSD

**Идет с первого(большего) разряда**

_Будем считать, что у всех элементов одинаковое число разрядов. Если это
 не так, то положим на более старших разрядах элементы с самым маленьким значением — для чисел это 0. Сначала исходный массив делится на k частей, где k — основание, выбранное для представления сортируемых объектов. Эти части принято называть "корзинами" или "карманами". В первую корзину попадают элементы, у которых старший разряд с номером d=0 имеет значение 0. Во вторую корзину попадают элементы, у которых старший разряд с номером d=0 имеет значение 1 и так далее. Затем элементы, попавшие в разные корзины, подвергаются рекурсивному разделению по следующему разряду с номером d=1. Рекурсивный процесс разделения продолжается, пока не будут перебраны все разряды сортируемых объектов и пока размер корзины больше единицы. То есть останавливаемся когда d>m или l⩾r, где m — максимальное число разрядов в сортируемых объектах, l, r — левая и правая границы отрезка массива A._
 _В основу распределения элементов по корзинам положен метод распределяющего подсчета элементов с одинаковыми значениями в сортируемом разряде. Для этого выполняется просмотр массива и подсчет количества элементов с различными значениями в сортируемом разряде. Эти счетчики фиксируются во вспомогательном массиве счетчиков cnt. Затем счетчики используются для вычисления размеров корзин и определения границ разделения массива. В соответствии с этими границами сортируемые объекты переносятся во вспомогательный массив c, в котором размещены корзины. После того как корзины сформированы, содержимое вспомогательного массива c переносится обратно в исходный массив A и выполняется рекурсивное разделение новых частей по следующему разряду в пределах границ корзин, полученных на предыдущем шаге._

**Изначально запускаем функцию так radixSort(A,0,A.length−1,1)**

```cpp
 function msd_radixSort(int[] A, int l, int r, int d):
     if d > m or l >= r 
         return
     for j = 0 to k + 1 
         cnt[j] = 0
     for i = l to r                              
         j = digit(A[i], d)
         cnt[j + 1]++
     for j = 2 to k
         cnt[j] += cnt[j - 1]
     for i = l to r
         j = digit(A[i], d)
         c[l + cnt[j]] = A[i]
         cnt[j]--
     for i = l to r
         A[i] = c[i]
     radixSort(A, l, l + cnt[0] - 1, d + 1)
     for i = 1 to k
         radixSort(A, l + cnt[i - 1], l + cnt[i] - 1, d + 1)
```

```cpp
void sift_down(int arr[], int i, int size) {
    while (i * 2 + 1 < size) {
        int left = i * 2 + 1;
        int right = i * 2 + 2;
        int j = left;
        if (right < size && arr[right] < arr[left]) {
            j = right;
        }
        if (arr[i] <= arr[j]) break;
        swap(arr[i], arr[j]);
        i = j;
    }
}

void sift_up(int arr[], int i) {
    while (i > 0) {
        int j = (i - 1) / 2;
        if (arr[j] <= arr[i]) break;
        swap(arr[i], arr[j]);
        i = j;
    }
}

void insert(int min_heap[], int& size, int num) {
    min_heap[size] = num;
    sift_up(min_heap, size);
    size++;
}

int extract_min(int min_heap[], int& size) {
    int min = min_heap[0];
    swap(min_heap[0], min_heap[size - 1]);
    sift_down(min_heap, 0, size);
    return min;
}

//Имеет второй вариант через 2 кучи(рассказать)
void delete_from_h(int arr[], int i, int& size) {
    swap(arr[i], arr[size - 1]);
    size--;
    sift_down(arr, i, size);
    sift_up(arr, i);
}
```

```cpp
void heap_sort(int arr[], int n) {
    //Строим кучу за O(n)
    for (int i = n / 2; i >= 0; i--) {
        sift_down(arr, i, n);
    }
    //Сортировка за O(nlogn)
    int size = n;

}
```

### Основные операции над списками

#### **Вставка**

_Очевиден случай, когда необходимо добавить элемент (newHead) в голову списка. Установим в этом элементе ссылку на старую голову, и обновим указатель на голову._

```cpp
 function insert(Node thatElement):
   thatElement.next = thisElement.next 
   thisElement.next = thatElement
```

![alt text](image-8.png)

---

#### **Поиск**

_Для того, чтобы найти элемент по значению (value), будем двигаться по списку от головы до конца и сравнивать значение в элементах с искомым. Если элемента в списке нет, то возвращаем NULL._

```cpp
 Node search(int value):
   node = head
   while node != NULL and value != node.value
       node = node.next
   return node
```

---

#### **Удаление**

_Для того, чтобы удалить голову списка, переназначим указатель на голову на второй элемент списка, а голову удалим._

```cpp
 function removeHead():
   if head != NULL
       tmp = head
       head = head.next
       delete tmp
```

![alt text](image-9.png)

_Удаление элемента после заданного (thisElement) происходит следующим образом: изменим ссылку на следующий элемент на следующий за удаляемым, затем удалим нужный объект._

```cpp
 function removeAfter(Node thisElement):
   if thisElement.next != NULL
       tmp = thisElement.next
       thisElement.next = thisElement.next.next
       delete tmp
```

### Алгоритмы циклического списка

#### **Поиск цикла в списке**

_Для начала необходимо уметь определять — список циклический или нет. Воспользуемся алгоритмом Флойда "Черепаха и заяц". Пусть за одну итерацию первый указатель (черепаха) переходит к следующему элементу списка, а второй указатель (заяц) на два элемента вперед. Тогда, если эти два указателя встретятся, то цикл найден, если дошли до конца списка, то цикла нет._

```cpp
 boolean hasCycle(Node head):
   tortoise = head
   hare = head
   repeat
     if hare == NULL or hare.next == NULL 
       return false
     tortoise = tortoise.next
     hare = hare.next.next
   until tortoise == hare
   return true
```

_Если цикла не существует, то заяц первым дойдет до конца и функция возвратит false. В другом случае, в тот момент, когда и черепаха и заяц находятся в цикле, расстояние между ними будет сокращаться на 1, что гарантирует их встречу за конечное время._

---

#### **Поиск длины хвоста в списке с циклом**

_Так как для поиска хвоста мы должны знать, что цикл существует, воспользуемся предыдущей функцией и при выходе из неё запомним "момент встречи" зайца и черепахи. Назовем её pointMeeting._

---

##### **Реализация за O(n^2^)**

_Будем последовательно идти от начала цикла и проверять, лежит ли этот элемент на цикле. На каждой итерации запустим от pointMeeting вперёд указатель. Если он окажется в текущем элементе, прежде чем посетит pointMeeting снова, то точку окончания (начала) хвоста нашли._

---

##### **Реализация за O(nlogn)**

_Реализацию, приведенную выше можно улучшить. Для этого воспользуемся бинарным поиском. Сначала проверим голову списка, потом сделаем 2 шага вперёд, потом 4, потом 8 и так далее, пока не окажемся на цикле. Теперь у нас есть две позиции — на левой границе, где мы в хвосте, и на правой — в цикле. Сделаем бинарный поиск уже по этому отрезку и таким образом найдём цикл за O(nlogn)._

---

#### **Задача про обращение списка**

_Для того, чтобы обратить список, необходимо пройти по всем элементам этого списка, и все указатели на следующий элемент заменить на предыдущий. Эта рекурсивная функция принимает указатель на голову списка и предыдущий элемент (при запуске указывать NULL), а возвращает указатель на новую голову списка._

```cpp
 Node reverse(Node current, Node prev):
   if current == NULL
     return prev
   next = current.next
   current.next = prev
   return reverse(next, current)
```

_Алгоритм корректен, поскольку значения элементов в списке не изменяются, а все указатели next изменят свое направление, не нарушив связности самого списка._

## Стек и Очередь

### Стек (Stack) - LIFO

#### Операции и определение

_Стек (от англ. stack — стопка) — структура данных, представляющая из себя упорядоченный набор элементов, в которой добавление новых элементов и удаление существующих производится с одного конца, называемого вершиной стека. Притом первым из стека удаляется элемент, который был помещен туда последним, то есть в стеке реализуется стратегия «последним вошел — первым вышел» (last-in, first-out — LIFO). Примером стека в реальной жизни может являться стопка тарелок: когда мы хотим вытащить тарелку, мы должны снять все тарелки выше. Вернемся к описанию операций стека:_

* **empty** — проверка стека на наличие в нем элементов,
* **push (запись в стек)** — операция вставки нового элемента,
* **pop (снятие со стека)** — операция удаления нового элемента.
*  **top** -  получить значение с верхушки стека
  
![alt text](image-11.png)

---

| Параметр | Значение |
|----------|----------|
| **Принцип** | LIFO (Last-In-First-Out) |
| **Инвариант** | Последний добавленный элемент извлекается первым |
| **Основные операции** | Push, Pop, Top |
| **Память** | O(n) |

---

#### Асимптотика операций стека

| Операция | Сложность | Описание |
|----------|-----------|----------|
| **Push** | O(1) | Добавление элемента на вершину |
| **Pop** | O(1) | Удаление элемента с вершины |
| **Peek/Top** | O(1) | Просмотр вершины без удаления |
| **IsEmpty** | O(1) | Проверка на пустоту |
| **Search** | O(n) | Поиск элемента в стеке |

#### Реализации стека

##### 1. На массиве (статический)

_Перед реализацией стека выделим ключевые поля:__

_𝚜[𝟷…𝚗] — массив, с помощью которого реализуется стек, способный вместить не более n элементов,
𝚜.𝚝𝚘𝚙 — индекс последнего помещенного в стек элемента.
Стек состоит из элементов 𝚜[𝟷…𝚜.𝚝𝚘𝚙], где 𝚜[𝟷] — элемент на дне стека, а 𝚜[𝚜.𝚝𝚘𝚙] — элемент на его вершине. Если 𝚜.𝚝𝚘𝚙=𝟶, то стек не содержит ни одного элемента и является пустым (англ. empty). Протестировать стек на наличие в нем элементов можно с помощью операции — запроса 𝚜𝚝𝚊𝚌𝚔𝙴𝚖𝚙𝚝𝚢. Если элемент снимается с пустого стека, говорят, что он опустошается (англ. underflow), что обычно приводит к ошибке. Если значение 𝚜.𝚝𝚘𝚙 больше 𝚗, то стек переполняется (англ. overflow). (В представленном ниже псевдокоде возможное переполнение во внимание не принимается.)_

Каждую операцию над стеком можно легко реализовать несколькими строками кода:

```cpp
boolean empty():
  return s.top == 0

function push(element : T):
  s.top = s.top + 1
  s[s.top] = element

T pop():
  if empty()
    return error "underflow"
  else 
    s.top = s.top - 1
    return s[s.top + 1]

```

##### 2. На динамическом массиве

_Возможна реализация стека на динамическом массиве, в результате чего появляется существенное преимущество над обычной реализацией: при операции push мы никогда не сможем выйти за границы массива, тем самым избежим ошибки исполнения._

_Создадим вектор и определим операции стека на нём. В функции 𝚙𝚞𝚜𝚑 Перед тем, как добавить новый элемент, будем проверять, не нужно ли расширить массив вдвое, а в 𝚙𝚘𝚙, перед тем, как изъять элемент из массива, — не нужно ли вдвое сузить размер вектора. Ниже приведён пример реализации на векторе._

**Ключевые поля:**

𝚜[𝟶…𝚗−𝟷] — старый массив, в котором хранится стек,
𝚗𝚎𝚠𝚂𝚝𝚊𝚌𝚔[𝟶…𝚗𝚎𝚠𝚂𝚒𝚣𝚎] — временный массив, где хранятся элементы после перекопирования,
𝚑𝚎𝚊𝚍 — верхушка стека,
𝚌𝚊𝚙𝚊𝚌𝚒𝚝𝚢 — размер массива.

```cpp
function push(element : T):
  if head == capacity - 1
    T newStack[capacity * 2]
    for i = 0 to capacity - 1
      newStack[i] = s[i]
    s = newStack
    capacity = capacity * 2
  head++
  s[head] = element

T pop():
  temp = s[head]
  head--
  if head < capacity / 4
    T newStack[capacity / 2]
    for i = 0 to capacity / 4 - 1
      newStack[i] = s[i]
    s = newStack
    capacity = capacity / 2
  return temp
```

##### 3. На связном списке

_Стек можно реализовать и на списке. Для этого необходимо создать список и операции работы стека на созданном списке. Ниже представлен пример реализации стека на односвязном списке. Стек будем "держать" за голову. Добавляться новые элементы посредством операции 𝚙𝚞𝚜𝚑 будут перед головой, сами при этом становясь новой головой, а элементом для изъятия из стека с помощью 𝚙𝚘𝚙 будет текущая голова. После вызова функции 𝚙𝚞𝚜𝚑 текущая голова уже станет старой и будет являться следующим элементом за добавленным, то есть ссылка на следующий элемент нового элемента будет указывать на старую голову. После вызова функции 𝚙𝚘𝚙 будет получена и возвращена информация, хранящаяся в текущей голове. Сама голова будет изъята из стека, а новой головой станет элемент, который следовал за изъятой головой._

**Заведем конструктор вида ListItem(next : ListItem, data : T)**

**Ключевые поля:**

𝚑𝚎𝚊𝚍.𝚍𝚊𝚝𝚊 — значение в верхушке стека,
𝚑𝚎𝚊𝚍.𝚗𝚎𝚡𝚝 — значение следующее за верхушкой стека.

_В реализации на списке, кроме самих данных, хранятся указатели на следующие элементы, которых столько же, сколько и элементов, то есть, так же 𝚗. Стоит заметить, что стек требует O(n) дополнительной памяти на указатели в списке._

```cpp
function push(element : T):
  head = ListItem(head, element)

T pop():
  data = head.data
  head = head.next
  return data
```

---

#### **Реальный код:**

```cpp
#include <iostream>

// Узел стека
struct StackNode {
    int data;        // Данные
    StackNode* next; // Указатель на следующий узел
};

// Структура стека
struct Stack {
    StackNode* top;  // Вершина стека
};

// Создание пустого стека
void createStack(Stack* stack) {
    stack->top = NULL;
}

// Проверка на пустоту
bool isEmpty(Stack* stack) {
    return stack->top == NULL;
}

// Добавление элемента в стек
void push(Stack* stack, int value) {
    // Создаем новый узел
    StackNode* newNode = new StackNode;
    newNode->data = value;
    
    // Новый узел становится вершиной
    newNode->next = stack->top;
    stack->top = newNode;
}

// Удаление элемента из стека
void pop(Stack* stack) {
    if (isEmpty(stack)) {
        std::cout << "Стек пуст!" << std::endl;
        return;
    }
    
    // Запоминаем текущую вершину
    StackNode* temp = stack->top;
    
    // Перемещаем вершину на следующий элемент
    stack->top = stack->top->next;
    
    // Удаляем старую вершину
    delete temp;
}

// Просмотр вершины стека
int peek(Stack* stack) {
    if (isEmpty(stack)) {
        std::cout << "Стек пуст!" << std::endl;
        return -1; // Возвращаем -1 при ошибке
    }
    return stack->top->data;
}

// Вывод стека
void printStack(Stack* stack) {
    if (isEmpty(stack)) {
        std::cout << "Стек пуст!" << std::endl;
        return;
    }
    
    std::cout << "Стек (сверху вниз): ";
    StackNode* current = stack->top;
    while (current != NULL) {
        std::cout << current->data << " ";
        current = current->next;
    }
    std::cout << std::endl;
}

// Очистка стека
void clearStack(Stack* stack) {
    while (!isEmpty(stack)) {
        pop(stack);
    }
}

// Подсчет размера стека
int getSize(Stack* stack) {
    int count = 0;
    StackNode* current = stack->top;
    while (current != NULL) {
        count++;
        current = current->next;
    }
    return count;
}
```

### Очередь (Queue) - FIFO

#### Определение и операции

_Очередь (англ. queue)  — это структура данных, добавление и удаление элементов в которой происходит путём операций 𝚙𝚞𝚜𝚑 и 𝚙𝚘𝚙 соответственно. Притом первым из очереди удаляется элемент, который был помещен туда первым, то есть в очереди реализуется принцип «первым вошел — первым вышел» (англ. first-in, first-out — FIFO). У очереди имеется голова (англ. head) и хвост (англ. tail). Когда элемент ставится в очередь, он занимает место в её хвосте. Из очереди всегда выводится элемент, который находится в ее голове. Очередь поддерживает следующие операции:_

1. 𝚎𝚖𝚙𝚝𝚢 — проверка очереди на наличие в ней элементов,
2. 𝚙𝚞𝚜𝚑 (запись в очередь) — операция вставки нового элемента,
3. 𝚙𝚘𝚙 (снятие с очереди) — операция удаления нового элемента,
4. 𝚜𝚒𝚣𝚎 — операция получения количества элементов в очереди.
5. _frant_ - получение значения с начала очереди

![alt text](image-12.png)

| Параметр | Значение |
|----------|----------|
| **Принцип** | FIFO (First-In-First-Out) |
| **Инвариант** | Первый добавленный элемент извлекается первым |
| **Основные операции** | Enqueue, Dequeue, Front |
| **Память** | O(n) |

| **Front** | O(1) | Просмотр первого элемента |
| **IsEmpty** | O(1) | Проверка на пустоту |
| **Search** | O(n) | Поиск элемента в очереди |

##### 1. Циклическая очередь(на массиве)

_Очередь, способную вместить не более 𝚗 элементов, можно реализовать с помощью массива 𝚎𝚕𝚎𝚖𝚎𝚗𝚝𝚜[𝟶…𝚗−𝟷]. Она будет обладать следующими полями:_
𝚑𝚎𝚊𝚍 — голова очереди,
𝚝𝚊𝚒𝚕 — хвост очереди.

```cpp
boolean empty():
  return head == tail

function push(x : T):
  if (size() != n)
    elements[tail] = x
    tail = (tail + 1) % n

T pop():
  if (empty()) 
    return null
  x = elements[head]
  head = (head + 1) % n
  return x

int size()
  if head > tail
    return n - head + tail
  else
    return tail - head
```

## Дек (Deque) - двусторонняя очередь

### Определение и операции

| Параметр | Значение |
|----------|----------|
| **Полное название** | Double Ended Queue |
| **Принцип** | Добавление и удаление с обоих концов |
| **Инвариант** | Поддерживает порядок элементов с двусторонним доступом |
| **Память** | O(n) |
| **Гибрид** | Объединяет возможности стека и очереди |

_Дек (от англ. deque — double ended queue) — структура данных, представляющая из себя список элементов, в которой добавление новых элементов и удаление существующих производится с обоих концов. Эта структура поддерживает как FIFO, так и LIFO, поэтому на ней можно реализовать как стек, так и очередь. В первом случае нужно использовать только методы головы или хвоста, во втором — методы push и pop двух разных концов. Дек можно воспринимать как двустороннюю очередь. Он имеет следующие операции:_

𝚎𝚖𝚙𝚝𝚢 — проверка на наличие элементов,
𝚙𝚞𝚜𝚑𝙱𝚊𝚌𝚔 (запись в конец) — операция вставки нового элемента в конец,
𝚙𝚘𝚙𝙱𝚊𝚌𝚔 (снятие с конца) — операция удаления конечного элемента,
𝚙𝚞𝚜𝚑𝙵𝚛𝚘𝚗𝚝 (запись в начало) — операция вставки нового элемента в начало,
𝚙𝚘𝚙𝙵𝚛𝚘𝚗𝚝 (снятие с начала) — операция удаления начального элемента.

![alt text](image-14.png)

---

| Операция | Сложность | Описание |
|----------|-----------|----------|
| **𝚙𝚞𝚜𝚑𝙵𝚛𝚘𝚗𝚝** | O(1) | Добавление в начало |
| **𝚙𝚞𝚜𝚑𝙱𝚊𝚌𝚔** | O(1) | Добавление в конец |
| **𝚙𝚘𝚙𝙵𝚛𝚘𝚗𝚝** | O(1) | Удаление из начала |

#### 1. На двусвязном списке

**Ключевые поля:**

ListItem(data : T, next : ListItem, prev : ListItem) — конструктор,
𝚝𝚊𝚒𝚕 — ссылка на хвост,
𝚑𝚎𝚊𝚍 — ссылка на голову.

---

_Дек очень просто реализуется на двусвязном списке. Он состоит из элементов 𝚑𝚎𝚊𝚍…𝚝𝚊𝚒𝚕. Элементы всегда добавляются либо в 𝚝𝚊𝚒𝚕.𝚙𝚛𝚎𝚟, либо в 𝚑𝚎𝚊𝚍.𝚗𝚎𝚡𝚝. В данной реализации не учитывается изъятие из пустого дека._

```cpp
function initialize():
  head = ListItem(null, null, null)
  tail = ListItem(null, null, head)
  head.next = tail

function pushBack(x : T):
  head = ListItem(x, head, null)
  head.next.prev = head

T popBack():
  data = head.data
  head = head.next
  return data

function pushFront(x : T):
  tail = ListItem(x, null, tail)
  tail.prev.next = tail

T popFront():
  data = tail.data
  tail = tail.prev
  return data

```

#### 2. Цикличсекий дек (массив)

_Во всех циклических реализациях изначально присвоены следующие значения 𝚑𝚎𝚊𝚍=𝟶 и 𝚝𝚊𝚒𝚕=𝟶. Ключевые поля:_

_𝚍[𝟶…𝚗−𝟷] — массив, с помощью которого реализуется дек, способный вместить не более n элементов,
𝚑𝚎𝚊𝚍 — индекс головы дека,
𝚝𝚊𝚒𝚕 — индекс хвоста.
Дек состоит из элементов 𝚍[𝚑𝚎𝚊𝚍…𝚝𝚊𝚒𝚕−𝟷] или 𝚍[𝟶…𝚝𝚊𝚒𝚕−𝟷] и 𝚍[𝚑𝚎𝚊𝚍…𝚗−𝟷]. Всего он способен вместить не более n элементов. В данной реализации учитывается переполнение и правильно обрабатывается изъятие из пустого дека. Недостатком является константная длина массива, хранящего элементы. Все операции выполняются за O(1)._

```cpp
function pushBack(x : T):
  if (head == (tail + 1) % n)
    return error "overflow"
  d[tail] = x
  tail = (tail + 1) % n

T popBack():
  if (empty()) 
    return error "underflow" 
  tail = (tail - 1 + n) % n
  return d[tail]

function pushFront(x : T):
  if (head == (tail + 1) % n)
    return error "overflow"
  head = (head - 1 + n) % n
  d[head] = x

T popFront():
  if (empty()) 
    return error "underflow" 
  T ret = d[head]
  head = (head + 1) % n
  return ret
```

### Реализации и их особенности

#### 1. Классический бинарный поиск

```cpp
bool binary_search(int arr[], int n, int x) {
    int l = 0;
    r = n - 1;
    while (l <= r) {
        int mid = l + (r - l) / 2;
        if (arr[mid] == x) return true;
        else if (arr[mid] < x) l = mid + 1;
        else r = mid - 1;
    }
    return false;
}
```

```cpp
    if (i & j) return false;

    int cnt = 0;
    
    for (int k = 0; k < W; k++) {
        if ((combined & (1 << k)) == 0) {
            cnt++;
        } else {
            if (cnt % 2 != 0) return false;
            cnt = 0;
        }
    }
    return (cnt % 2 == 0);
}

int main() {

    int W, H;
    std::cin >> W >> H;
    int64_t dp[1 << W][1 << W];
    for (size_t i = 0; i < (1 << W); i++) {
        for (size_t j = 0; j < (1 << W); j++) {
            if (check(i, j, W)) {
                dp[i][j] = 1;
            } else {
                dp[i][j] = 0;
            }
        }
    }

    int64_t arr[H + 1][1 << W];
    for (int k = 0; k < H + 1; k++) {
        for (size_t i = 0; i < (1 << W); i++) {
            arr[k][i] = 0;
        }
    }
    arr[0][0] = 1;
    for (int k = 1; k < H + 1; k++) {
        for (size_t i = 0; i < (1 << W); i++) {
            for (size_t j = 0; j < (1 << W); j++) {
                arr[k][i] += arr[k - 1][j] * dp[j][i];
            }
        }
    }
    
    std::cout << arr[H][0];
}
```

## Персистентность

### Определение

**Персистентность** _- это св-во структуры хранить свои предыдцщие версии и изменения._

---

###  Уровни персистентности

- **Частичная (англ. partial)** - _к каждой версии можно делать запросы, но изменять можно только последнюю версию структуры данных._
- **Полная (англ. full)** - _можно менять не только последнюю, но и любую версию структур данных, также к любой версии можно делать запросы._
- **Конфлюэнтная (англ. confluent)** - _позволяют объединять две структуры данных в одну (деревья поиска, которые можно сливать)._
- **Функциональная (англ. functional)** - _полностью персистентны по определению, так как в них запрещаются уничтожающие присваивания, т.е. любой переменной значение может быть присвоено только один раз и изменять значения переменных нельзя._

---

### Способы преобразования структур данных в персистентные

#### Полное копирование

**Полное копирование** - _самый неэффектиынй способ, когда при любой операции изменения полностью копируется структура данных и в получившуюся новую копию вносятся изменения._

---

#### Копирование пути

**Копирование пути** - _значительно более эффективный способ сделать структуру персистентной. Чаще всего применяется в структурам, представляющим деревья._

**Рассмотрим на примере ДО:**

_Для построения персистентного дерева отрезков из n элементов необходимо применить n раз операцию добавления элемента к последней версии дерева. Для того, чтобы добавить новый элемент к k-ой версии дерева, необходимо проверить, является ли оно полным бинарным. Если да, то создадим новый корень, левым сыном сделаем roots[k]. Иначе, сделаем копию корня исходной версии. Добавим корень в конец массива корней. Далее, спускаясь от корня к первому свободному листу, будем создавать несуществующие узлы и клонировать существующие. После этого в новой ветке необходимо обновить значение функции и некоторые указатели дочерних элементов. Поэтому, возвращаясь из рекурсии, будем менять один указатель на только что созданную или скопированную вершину, а также обновим значение функции, для которой строилось дерево. После этой операции в дереве появится новая версия, содержащая вставленный элемент._

_Для того, чтобы изменить элемент в персистентном дереве отрезков, необходимо сделать следующие действия: спустимся в дереве от корня нужной версии до требуемого элемента, скопируем его, изменим значение, и, поднимаясь по дереву, будем клонировать узлы. При этом необходимо менять указатель на одного из детей на узел, созданный при предыдущем клонировании. После копирования корня, добавим новый корень в конец массива корней._

![alt text](image-44.png)

_Здесь изображено персистентное дерево отрезков с операцией минимум, в котором изначально было 3 вершины. Сперва к нему была добавлена вершина со значением 2, а потом изменена вершина со значением 7. Цвет ребер и вершин соответствует времени их появления. Синий цвет элементов означает, что они были изначально, зеленый - что они появились после добавления, а оранжевый - что они появились после изменения элемента._

**На каждую версию такого персистентного ДО уйдет около O(logn) памяти, т.к. храним тольку одну измененную ветку и указатели на предыдущую неизмененнную версию дерева**

**На примере персистентного стека:**
_Будем хранить массив указателей на текующую голову стека(верхний эл-т). После каждого изменения(pop/push) добавляем в этот массив обновленную голову, тогда если мы захотим вернуться к какой-либо версии, то нам нужно будет всего лишь взять указатель на голову в этой версии, а от него уже можно будет получить весь стек этой версии_

---

#### Хранения изменений

**Хранение изменений(или метод "толстых" узлов)** - _похоже на полное копирование, но более эффективно, т.к. мы храним не всю структуру, а только изменяющиеся от версии к версии части._

_На примере **персистентного массива** мы будем после каждой операции **update** сохранять пару **(version, new_value)**  в массив измменений каждого элемента(pos)._

```cpp
Для массива:
[2, 6 ,7, 0] -> [2, 13, 7, 0] -> [2, 13, 3, 0] -> [2, 8, 3, 0]

pos 0: [] -> [] -> [] -> []
pos 1: [] -> [(1, 13)] -> [(1, 13), (3, 8)]
pos 2: [] -> [] -> [(2, 3)] -> [(2, 3)]
pos 3: [] -> [] -> [] -> []
```

## Деревья

### Вариации деревьев

#### Дерево отрезков

##### Построение за O(n) и асимптотики

_Выделяют два основных способа построения дерева отрезков: **построение снизу и построение сверху**. При построении снизу алгоритм поднимается от листьев к корню (Просто начинаем заполнять элементы массива t от большего индекса к меньшему, таким образом при заполнении элемента i его дети 2i+1 и 2i+2 уже будут заполнены, и мы с легкостью посчитаем бинарную операцию от них), а при построении сверху спускается от корня к листьям. Особенные изменения появляются в реализации запросов к таким деревьям отрезков._

![alt text](image-40.png)

| Операция | Временная сложность | Пространственная сложность (структура) |
|----------|-------------------|----------------------------------------|
| Построение (build) | O(n) | O(n) |
| Запрос (query) | O(log n) | O(1)* |
| Обновление точки (point update) | O(log n) | O(1)* |
| Обновление отрезка (lazy propagation) | O(log n) | O(n) |
| Обновление отрезка (без lazy) | O(n) | O(n) |

**Примечания:**

1. **Простраственная сложность O(n)** для хранения дерева — это **4n** в худшем случае для рекурсивной реализации на массиве. Итеративная реализация может уменьшить до **2n** (или ~2n).
2. **O(1)** по памяти для запроса и точечного обновления — дополнительная память, не считая хранения самого дерева (т.е. стек вызовов или локальные переменные).
3. **Lazy propagation** позволяет делать обновление на отрезке за O(log n), но увеличивает затраты по памяти вдвое (обычно ещё массив размера 4n).
4. Для **персистентного дерева отрезков** временные сложности те же O(log n), но пространственная O(log n) на операцию (или O(m log n) для m версий).

**O(4n)** — стандартный запас для рекурсивного дерева на массиве (хотя можно ужать до 2n, но с потерей простоты кода).

##### Бинарный спуск

_В задаче, где необходимо найти при известной левой границе l минимальную правую границу r такую, что на этом подотрезке сумма >= s, можно применить **бинарный спуск**._

**Бинарный спуск = БинПоиск + ДО**

![alt text](image-41.png)

**_Алгоритм:_**

- Рекурсивно идем в левое поддерево
- Если сумма в нем >= s - пробуем спускаться ниже.
- Иначе сохраняем это поддерево, вычитаем из s сумму в поддереве.
- Аналогично идем в правое поддерево, ищем в нем новое s

**Асимптотика: O(logn)**_, т.к. алгоритм раздваивается на каждом шаге_

##### Отложенные операции. Групповое обновление

_Рассмотрим в общем виде реализацию массовой операции на отрезке. Пусть необходимо отвечать на запросы относительно операции ⊕, а запрос массового обновления идет по операции ⊙._

_Для эффективной реализации будем использовать описанную выше структуру — несогласованные поддеревья. В каждой вершине, помимо непосредственно результата выполнения операции ⊕, храним несогласованность — величина, с которой нужно выполнить операцию ⊙ для всех элементов текущего отрезка. Тем самым мы сможем обрабатывать запрос массового обновления на любом подотрезке эффективно, вместо того чтобы изменять все O(N) значений. Как известно из определения несогласованных поддеревьев, в текущий момент времени не в каждой вершине дерева хранится истинное значение, однако когда мы обращаемся к текущему элементу мы работаем с верными данными. Это обеспечивается "проталкиванием" несогласованности детям (процедура push) при каждом обращений к текущей вершине. При этом после обращения к вершине необходимо пересчитать значение по операции ⊕, так как значение в детях могло измениться._

**В каждом нижеприведенном псевдокоде в узлах дерева хранятся структуры из четырех полей:**

- 𝚕𝚎𝚏𝚝 — левая граница полуинтервала, за который "отвечает" текущая вершина.
- 𝚛𝚒𝚐𝚑𝚝 — правая граница этого полуинтервала.
- 𝚊𝚗𝚜 — результат на отрезке по операции ⊕.
- 𝚍 — несогласованность.

---

`push`
_**"Проталкивание" несогласованности детям.** Необходимо выполнять как только идет рекурсивный запуск от текущей вершины к её детям. Нужно это для того, чтобы в детях в момент обработки были корректные данные._

```cpp
void push(int node) {                                               // node - текущая вершина 
       tree[2 * node + 1].d = tree[2 * node + 1].d ⊙ tree[node].d;
       tree[2 * node + 2].d = tree[2 * node + 2].d ⊙ tree[node].d;
       tree[node].d = ⊥;                                           // Нейтральный элемент 
}
```

---

`update`
_**Процедура обновления на отрезке.** Данная процедура выполняет разбиение текущего отрезка на подотрезки и обновление в них несогласованности. **Очень важно выполнить push как только идет рекурсивный вызов от детей**, чтобы избежать некорректной обработки в детях. И так как значение в детях могло измениться, то необходимо выполнить обновление ответа по операции ⊕ на текущем отрезке._

```cpp
void update(int node, int a, int b, T val) {
    // val - значение, которое поступило в качестве параметра на запрос, a и b - границы запроса 
       l = tree[node].left;
       r = tree[node].right; 
       if  [l, r) ∩ [a, b) == ∅
           return;
       if [l, r) ⊂ [a, b)
           tree[node].d = tree[node].d ⊙ val;
           return;

       push(node); 
   // Обновление детей 
       update(2 * node + 1, a, b, val);
       update(2 * node + 2, a, b, val);
    // Пересчет значения на текущем отрезке 
       tree[node].ans = (tree[2 * node + 1].ans
       ⊙ tree[2 * node + 1].d) ⊕ (tree[2 * node + 2].ans 
       ⊙ tree[2 * node + 2].d);
}
```

---

`query`
_**Получение ответа по операции ⊕.** Отличие от операции обновления лишь в том, что **для каждого отрезка разбиения необходимо не обновить несогласованность, а сложить по операции ⊕ с текущим ответом истинное значение на отрезке** (то есть обновить значение в каждой вершине после update и пересчитать query для подотрезка)._

```cpp
T query(int node, int a, int b) {
       l = tree[node].left;
       r = tree[node].right; 
       if  [l, r ) ∩ [a, b) == ∅
           return ⊥;
       if [l, r) ⊂ [a, b)
           return tree[node].ans ⊙ tree[node].d;
       push(node);   
       T ans = query(node * 2 + 1, a, b) ⊕  
                query(node * 2 + 2, a, b));
       tree[node].ans = (tree[2 * node + 1].ans ⊙ tree[2 * node + 1].d) ⊕ 
                         (tree[2 * node + 2].ans ⊙ tree[2 * node + 2].d);
       return ans;
}
```

##### Динамическое ДО(неявное)

###### Определение и асимптотика

_Вместо того, чтобы хранить массив на 4n эл-ов, храним дерево в виде списка, где в каждой ноде указатель на 2 сыновей, границы подотрезка и значение на нем._
_Каждый раз при запросе update строим только ту ветку, по которой хотим пройти, обновляем на ней значения. Ту часть, которую не используем - не строим и считаем, что она заполнена нулями(нейтральными эл-ми). Далее обращаясь к дереву, если необзодимая ветка уже есть - используем ее, если нет, то строим необходимую. Таким образом, экономим на памяти_

**Асимптотика остается такой же:**
`O(logn) + O(logn) (рекурсивное выделение памяти) = O(logn)`

###### Оффлайн запросы. Сжатие координат

_Предположим, что заранее знаем все запросы к нашей структуре. Выделим оттуда все запросы, в которых происходили изменения структуры и отбросим все остальные. Теперь переприсвоим нашим выделенным запросам новые координаты(< исходыных), отсортируем и построим на них обычное ДО._
_Таким образом, мы отбрасываем существование запросов, которые ничего не меняли(т.к. знаем все запросы заранее и что они не поменяются) и уменьшаем кол-во необходимой памяти_

![alt text](image-43.png)

**Асимптотика:**
`O(qlog(q)) - по времени,`
`O(q) - по памяти,`
`где q - кол-во запросов`

#### Дерево Фенвика

##### Определение

**Дерево Фе́нвика (англ. Binary indexed tree)** — _структура данных, требующая O(n) памяти и позволяющая эффективно (за O(logn)) выполнять следующие операции:_

- _изменять значение любого элемента в массиве_
- _выполнять некоторую ассоциативную, коммутативную, обратимую операцию ∘ на отрезке [i,j]._

_Пусть дан массив A=[a_0,a_1,...,a_n−1]. Деревом Фенвика будем называть массив T из n элементов: Ti=∑k=F(i)iak, где i=0..n−1 и F(i) — некоторая функция, от выбора которой зависит время работы операций над деревом. Рассмотрим функцию, позволяющую делать операции вставки и изменения элемента за время O(logn). Она задается простой формулой: **F(i)=i&(i+1)**, где & — это операция побитового логического AND. При AND числа и его значения, увеличенного на единицу, мы получаем это число без последних подряд идущих единиц._

- _**Для пересчёта дерева Фенвика** при изменении величины a_k необходимо **изменить элементы дерева T_i** для индексов i, для которых верно неравенство **F(i)⩽k⩽i**._

- _Все такие i, для которых меняется T_i при изменении a_k, можно найти по формуле **i_next=i_prev | (i_prev+1)**, где ∣ — это операция побитового логического OR._

---

##### Алгоритмы

###### Update

```cpp
function update(i, d):
   while i < N
       t[i] += d
       i = i | (i + 1)
```

###### Set

**Часто можно встретить задачу, где требуется заменить значение элемента a_i на x. Заметим, что если вычислить разность x и a_i, то можно свести эту задачу к операции прибавления d к a_i:**

```cpp
function set(i, x):
   d = x - a[i]
   a[i] = x
   update(i, d)
```

###### Build

**Построение дерева можно осуществить, исходя из его описания. Но можно быстрее, если использовать функцию update для каждого элемента массива A. Тогда мы получим время работы O(nlogn).**

```cpp
function build():
   for i = 0 to N - 1
       update(i, a[i])
```

**Получение префиксных сумм на отрезке:**

```cpp
int sum(i):

       result += t[i]
```

##### Асимптотика

| Параметр | Значение / Описание |
| :--- | :--- |
| **Инвариант** | Структура на основе бинарного представления индексов. Каждый элемент `tree[i]` хранит сумму элементов исходного массива на отрезке `[i - g(i) + 1, i]`, где `g(i) = i & (-i)` — младший ненулевой бит (операция `i & -i`). |
| **Назначение** | Эффективное вычисление **префиксных сумм** и **точечных обновлений** для мультимножества чисел. |
| **Пространственная сложность** | `O(n)` (использует массив размера `n+1`, индексация с 1) |
| **Инициализация (построение)** | `O(n log n)` (n раз `update`) |
| **Обновление элемента (point update)** | `O(log n)` |
| **Запрос префиксной суммы (prefix sum)** | `O(log n)` |
| **Запрос суммы на отрезке [l, r]** | `O(log n)` (как `sum(r) - sum(l-1)`) |
| **Точечный запрос значения a[i]** | `O(log n)` (через разность префиксных сумм) или `O(1)` если хранить исходный массив отдельно |
| **Обновление диапазона, точечный запрос** | Реализуется через **два дерева Фенвика** (разностный массив) |

---

**Ключевые особенности:**

1. **Проще реализации**, чем дерево отрезков, особенно для задач на префиксные суммы.
2. **Операция на отрезке, для которой строится дерево Фенвика, должна быть обратимой**, а это значит, что минимум (как и максимум) на отрезке это дерево считать не может, в отличие от дерева отрезков. Но если нам требуется найти минимум на префиксе, то дерево Фенвика справится с этой задачей. Такое дерево Фенвика поддерживает операцию уменьшения элементов массива. Пересчёт минимума в дереве происходит быстрее, чем обновление массива минимумов на префиксе.
3. **Константа** в асимптотике очень мала благодаря простым битовым операциям.
4. **Неявная структура** — не требует хранения дерева в явном виде (узлов и связей), только массив.

```cpp
            return r;
```

#### AVL дерево

##### Описание и асимптотика

**AVL дерево** - _вариация наивного BST, в котором за счет балансирования дерева получается свести асимптотику от **O(log H) до честного O(log n)**, т.е. избежать бамбука._

| Параметр | Значение / Описание |
| :--- | :--- |
| **Инвариант** | BST-инвариант и  **\|h(L)−h(R)\| ⩽1** для каждой вершины |
| **Пространственная сложность** | `O(n)` |
| **Вставка** | `O(log n)` |
| **Удаление** | `O(log n)` |
| **Поиск** | `O(log n)` |
| **Балансировка (повороты)** | `O(1)` на узел |
| **Построение из n элементов** | `O(n log n)` |

##### Структура узла

```cpp
struct Node {
    T key; //значение в вершине
    Node* left; // левый потомок
    Node* right; // правый потомок
    Node* parent; // указатель на родителя(необязателен)
    int height = 0;
};
```

##### Балансировка

![alt text](image-50.png)

```cpp
int get_height(const Node* root) { // посмотреть высоту дерева
    return root ? root->height : 0;
}

int get_balance(const Node* root) { // считаем h(L)−h(R) для двух поддеревьев
    return root ? get_height(root->left) - get_height(root->right) : 0;
}

void fix_height(Node*& root) { // обновление поля height в узле
    if (root) {
        int h1 = get_height(root->left);
        int h2 = get_height(root->right);
        root->height = std::max(h1, h2) + 1;
    }
}

Node* rotate_l(Node* root) { // малый левый поворот
    Node* b = root->right;
    root->right = b->left;
    b->left = root;
    fix_height(root);
    fix_height(b);
    return b;
}

Node* rotate_r(Node* root) { // малый правый поворот
    Node* b = root->left;
    root->left = b->right;
    b->right = root;
    fix_height(root);
    fix_height(b);
    return b;
}

Node* balance(Node* root) { // общая ф-ия для вращений
    if (!root) return root;
    fix_height(root);

    // Большой левый поворот
    if (get_balance(root) == -2) {
        if (get_balance(root->right) > 0) {
            root->right = rotate_r(root->right);
        }
        return rotate_l(root);
    }

    // Большой правый поворот
    if (get_balance(root) == 2) {
        if (get_balance(root->left) < 0) {
            root->left = rotate_l(root->left);
        }
        return rotate_r(root);
    }
    return root;
}
```

#### Splay дерево

##### Определение и асимптотика

**Splay-дерево** — _это двоичное дерево поиска. Оно позволяет находить быстрее те данные, которые использовались недавно, т.е. поднимает последний "потроганный" эл-т в корень._

| Параметр | Значение / Описание |
| :--- | :--- |
| **Инвариант** | BST-инвариант |
| **Пространственная сложность** | O(n) |
| **insert** | O^*^(log n) |
| **erase** | O^*^(log n) |
| **find** | O^*^(log n) |
| **splay** | O^*^(log n) |
| **merge** | O^*^(log n) |
| **split** | O^*^(log n) |
| **Построение** | O^*^(nlog n) |

_Операция **splay** делится на 3 случая различных поворотов:_

- zig
- zigzag
- zigzig

**_Ключевые особенности:_**

- _Splay(x) выполняется за **O(h)**, где h - длина пути от x до корня._

- _Доказывается, что splay(x) работает за **O^*^(logn)**._

- _**В среднем и худшем случаях** операции splay-дерева работают за **O^*^(logn)**, а в **лучшем** за **O(1)**._

- _В отличие от ранее разобранного AVL дерева в splay-дереве удобно реализуются операции **merge/split**._

---

##### Splay

###### zig

`Если p — корень дерева с сыном x, то совершаем один поворот вокруг ребра (x,p), делая x корнем дерева. Данный случай является крайним и выполняется только один раз в конце, если изначальная глубина x была нечетной.`

![alt text](image-52.png)

---

###### zigzag

`Если p — не корень дерева и x — левый ребенок, а p — правый, или наоборот, то делаем поворот вокруг ребра (x,p), а затем поворот нового ребра (x,g), где g — бывший родитель p.`

![alt text](image-55.png)

![alt text](image-56.png)

---

###### zigzig

`Если p — не корень дерева, а x и p — оба левые или оба правые дети, то делаем поворот ребра (p,g), где g отец p, а затем поворот ребра (x,p).`

![alt text](image-53.png)

![alt text](image-54.png)

###### Реализация splay

`Помимо указателей на левого и правого сыновей и ключа, будем хранить в каждой ноде указатель на родителя.`

```cpp
Node p(Node v): // получить родителя
  return v.parent

Node g(Node v): // получить родителя родителя
  return p(p(v))


void rotate_left(Node v): // левый поворот как в AVL
  Node p = p(v)
  Node r = v.right
  if (p != null)
    if (p.left == v)
      p.left = r
    else
      p.right = r
  Node tmp = r.left
  r.left = v
  v.right = tmp
  p(v) = r
  p(r) = p
  if (v.right != null)
    p(v.right) = v

// Симметрично правый поворот
// Необходимы для реализации zig, zigzig, zigzag


//Реализация splay:
void splay(Node v):
  while (p(v) != null)
    if (v == p(v).left) // если элемент - левый сын
      if (g(v) == null) // zig
        rotate_right(p(v))
      else if (p(v) == g(v).left) // zigzig
        rotate_right(g(v))
        rotate_right(p(v))
      else
        rotate_right(p(v)) //zigzag
        rotate_left(p(v))
    else // если элемент правый сын
      if (g(v) == null) // zig
        rotate_left(p(v))
      else if (p(v) == g(v).right) //zigzig
        rotate_left(g(v))
        rotate_left(p(v))
      else
        rotate_left(p(v)) // zigzag
        rotate_right(p(v))
```

##### Операции

###### merge

_Пусть **есть 2 splay-дерева T1 и T2**, таких что **все ключи в T1 меньше, чем все ключи в T2**. **Найдем наибольший(самый правый)** эл-т x в T1 и запустим **splay(x)**. Теперь, т.к. x наибольший в T1, у корня нет правого ребенка, тогда **подвесим в качестве правого ребенка T2**. Нетрудно заметить, что инвариант дерева поиска сохранится._

---

###### split

_Пусть мы **хотим разбить дерево по ключу k**. **Найдем  x - максимальный эл-т <= k** и запустим **splay(x)**. Теперь **в левом поддререве L все эл-ты <= x <= k**, а в **правом поддереве R все эл-ты > x > k**. Значит **возвращаем L и R**, что и требовалось._

---

###### find

_Стоит заметить, что операции работает точно так же, **как в наивном BST**, отличие состоит лишь в **вызове** операции **splay** после запроса find._

---

###### insert

_Сделаем **split(tree, x)**, разбив изначально дерево на 2 поддерева(в поддереве L все ключи < x, в R > x). Заметим, что **инвариант дерева поиска сохраняется** в таком случае, так что **подвесим L и R в качестве левого и правого детей x**._

---

###### erase

_Запустим **splay(x)**, тем самым **подняв x в корень**, теперь **удалим x** и сделаем **merge от его детей**._

#### Неявное дерево поиска

_Является **вариацией любого** дерева поиска, позволяющей работать с деревом как с массивом, т.е. выполянть следующие операции:_

- insert
- update
- delete
- find
- получить значение по индексу

_Будем в **каждом узле** хранить **количество эл-ов в его поддереве**, включая его самого. Тогда утверждается, что можно **однозначно восстановить индексы** эл-ов в исходном "массиве"._

### Обходы бинарного дерева

#### Виды обходов и асимптотика

| Обход | Порядок (узел `N`, лево `L`, право `R`) | Временная сложность | Пространственная сложность (стек) |
| :--- | :--- | :--- | :--- |
| **Inorder (симметричный)** | `L → N → R` | `O(n)` | `O(h)` (высота дерева) |
| **Preorder (прямой)** | `N → L → R` | `O(n)` | `O(h)` |
| **Postorder (обратный)** | `L → R → N` | `O(n)` | `O(h)` |

**Примечание:** _`h` — высота дерева. Для сбалансированного дерева `h = O(log n)`, для вырожденного `h = O(n)`._

#### Postorder Traversal

_Удобен для **удаления** эл-ов из дерева, т.к. обходит дерево **начиная с лисьтьев**_
